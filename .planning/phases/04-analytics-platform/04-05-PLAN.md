async function fetchData(append = false) {
  setLoading(true); setError(undefined);
  try {
    const response = await fetchAuditTrail(filters);
    if (append) {
      setEntries(prev => [...prev, ...response.entries]);
    } else {
      setEntries(response.entries);
    }

    setTotalCount(response.totalCount);

  } finally {
    setLoading(false);
  }
}

**State:**

const [alerts, setAlerts] = useState<AlertEvent[]>([]);
const [error, setError] = useState<string>();

**Polling effect:**
```typescript
useEffect(() => {
  let active = true;

  async function poll() {
    try {
      const response = await fetchPendingAlerts();
      if (!active) return;
      const newAlerts = response.alerts;
      if (newAlerts.length > 0) {
        setAlerts(prev => [...newAlerts, ...prev]);
      }
      setError(undefined);
    } catch (e) {
      if (!active) return;
      setError(e instanceof Error ? e.message : "Failed to fetch alerts");
    }
  }

  poll(); // Initial fetch
  const interval = setInterval(poll, 10000); // 10s per D-14

  return () => {
    active = false;
    clearInterval(interval);
  };
}, []);
```

**Alert display:**
```tsx
<section className="panel panel-alerts">
  <div className="panel-header">
    <h2>Alerts</h2>
    <span>{alerts.length > 0 ? `${alerts.length} alert${alerts.length !== 1 ? "s" : ""}` : "No alerts"}</span>
    {alerts.length > 0 && <button onClick={() => setAlerts([])}>Dismiss All</button>}
  </div>
  {error && <p className="error-banner">{error}</p>}
  {alerts.length === 0 ? (
    <p className="status-message">No active alerts</p>
  ) : (
    <ul className="alert-list">
      {alerts.map(alert => (
        <li key={alert.id} className={`alert-item alert-${alert.status}`}>
          <div className="alert-header">
            <strong className={alert.status === "firing" ? "text-danger" : "text-success"}>
              {alert.status === "firing" ? "FIRING" : "RESOLVED"}
            </strong>
            <span>{alert.ruleName}</span>
            <time>{new Date(alert.timestamp).toLocaleTimeString()}</time>
          </div>
          <p className="alert-message">{alert.message}</p>
          <button className="alert-dismiss" onClick={() => setAlerts(prev => prev.filter(a => a.id !== alert.id))}>
            Dismiss
          </button>
        </li>
      ))}
    </ul>
  )}
</section>
```
