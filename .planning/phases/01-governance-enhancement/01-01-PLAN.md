Create packages/contracts/src/governance/rule-engine.ts with the following Zod schemas:

1. RuleOperatorSchema: enum with "and", "or", "not"
2. ComparisonOperatorSchema: enum with "eq", "ne", "gt", "lt", "gte", "lte", "in", "contains", "startsWith", "endsWith"
3. RuleConditionSchema: object with:
   - field: string (dot-notation path like "complexity.total" or "sensitivity")
   - operator: ComparisonOperatorSchema
   - value: z.union([z.string(), z.number(), z.boolean(), z.array(z.string()), z.array(z.number())])
4. CompositeRuleSchema: recursive type using z.lazy() for nested rules:
   - operator: RuleOperatorSchema
   - rules: z.array(z.union([RuleConditionSchema, z.lazy(() => CompositeRuleSchema)]))
5. ApprovalRuleSchema: object with:
   - id: z.string().uuid()
   - name: z.string().min(1).max(200)
   - description: z.string().max(1000).optional()
   - version: z.number().int().positive().default(1)
   - versionStatus: z.enum(["draft", "published", "archived"]).default("draft")
   - conditions: z.union([RuleConditionSchema, CompositeRuleSchema])
   - actions: z.object({
       type: z.enum(["require_approval", "auto_approve", "auto_reject", "escalate"]),
       approvers: z.array(z.string()).optional(),
       escalationTarget: z.string().optional(),
       reason: z.string().optional()
     })
   - priority: z.number().int().min(0).max(1000).default(500)
   - enabled: z.boolean().default(true)
   - createdAt: z.string().datetime()
   - updatedAt: z.string().datetime()
   - createdBy: z.string().optional()

Export all type inferences using z.infer<typeof SchemaName>.

Add barrel export to packages/contracts/src/governance/index.ts.
</action>
