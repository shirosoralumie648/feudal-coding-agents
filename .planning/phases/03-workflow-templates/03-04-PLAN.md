```typescript
import type { FastifyInstance } from "fastify";
import { z } from "zod";

const SomeParamsSchema = z.object({ ... });

export function registerTaskRoutes(
  app: FastifyInstance,
  service: OrchestratorService = defaultOrchestratorService
) {
  app.get("/api/tasks", async () => service.listTasks());
  
  app.post("/api/tasks", async (request, reply) => {
    const payload = SomeSchema.parse({ ...request.body });
    const result = await service.createTask(payload);
    return reply.code(201).send(result);
  });
  
  app.get("/api/tasks/:taskId", async (request, reply) => {
    const params = SomeParamsSchema.parse(request.params);
    const task = await service.getTask(params.taskId);
    if (!task) return reply.code(404).send({ message: "Task not found" });
    return task;
  });
}
```

- `reply.code(201).send(result)` for POST creates
- `reply.code(404).send({ message: "..." })` for not-found
- Route paths: `/api/templates`, `/api/templates/:name`

    ===== Routes (implement all 10 endpoints) =====

    **1. POST /api/templates** -- Create a new template (as draft)
    - Parse body with `CreateTemplateInput`
    - Call `store.createTemplate(payload)`
    - Return 201 with the created template

    **2. GET /api/templates** -- List all templates
    - Optional query param `status` (draft | published)
    - Call `store.listTemplates(status ? { status } : undefined)`
    - Return 200 with array

    **3. GET /api/templates/:name** -- Get a template by name
    - Parse params with `TemplateNameParams`
    - Call `store.getTemplate(params.name)`
    - If not found, return 404 with `{ message: "Template not found" }`
    - Return 200 with template

    **4. PUT /api/templates/:name** -- Update a template
    - Parse params with `TemplateNameParams`
    - Parse body with `UpdateTemplateInput`
    - Read `if-match` header: `request.headers["if-match"]` -> parse to integer as expectedVersion
    - If no `if-match` header, return 400 with `{ message: "if-match header with event version is required for updates" }`
    - Call `store.updateTemplate(params.name, body, expectedVersion)`
    - If version mismatch error, return 409 with `{ message: error.message }`
    - Return 200 with updated template

    **5. POST /api/templates/:name/publish** -- Publish a draft template
    - Parse params with `TemplateNameParams`
    - Read `if-match` header for expectedVersion
    - Call `store.publishTemplate(params.name, expectedVersion)`
    - Return 200 with published template

    **6. POST /api/templates/:name/unpublish** -- Unpublish a template
    - Parse params with `TemplateNameParams`
    - Read `if-match` header for expectedVersion
    - Call `store.unpublishTemplate(params.name, expectedVersion)`
    - Return 200 with unpublished template

    **7. DELETE /api/templates/:name** -- Delete a draft template
    - Parse params with `TemplateNameParams`
    - Read `if-match` header for expectedVersion
    - Call `store.deleteTemplate(params.name, expectedVersion)`
    - Return 204 (no content)

    **8. GET /api/templates/:name/export** -- Export a published template as JSON
    - Parse params with `TemplateNameParams`
    - Call `store.exportTemplate(params.name)`
    - Return 200 with the TemplateExportPackage

    **9. GET /api/templates/:name/versions** -- Get version history
    - Parse params with `TemplateNameParams`
    - Call `store.getTemplateVersionHistory(params.name)`
    - Return 200 with version events array

    **10. POST /api/templates/:name/instantiate** -- Instantiate a task from a template
    - Parse params with `TemplateNameParams`
    - Parse body with `TemplateInstantiationSchema` (templateName, templateVersion, parameters)
    - Call `store.getTemplate(params.name)` -- if not found, 404
    - If template status is "draft", return 400 with `{ message: "Cannot instantiate a draft template. Publish it first." }`
    - Get the orchestrator service runtime via `defaultOrchestratorService` (imported from config.ts)
    - This route IS the integration point: it creates a task spec from the template instantiation parameters, then delegates to the orchestrator.
    - Return 201 with the instantiated task projection

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HTTP request -> API route | All external HTTP requests cross into route handlers -- Zod validates params/body at boundary |
| if-match header -> store | The eventVersion from the HTTP header crosses into store methods -- must be parsed as integer before use |
| Template import body -> store | External JSON package crosses into store via POST body -- format checked, internal state reset |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-15 | T (Tampering) | PUT /api/templates/:name -- missing if-match | mitigate | Route rejects updates without if-match header with 400; no blind writes allowed |
| T-03-16 | T (Tampering) | PUT /api/templates/:name -- stale if-match | mitigate | Optimistic locking via store throws on version mismatch, route returns 409 Conflict |
| T-03-17 | S (Spoofing) | POST /api/templates/:name/instantiate -- draft template | mitigate | Route rejects instantiation of draft templates with 400; only published templates can instantiate tasks |
| T-03-18 | D (Denial of Service) | POST /api/templates -- large body | accept | Zod schema validates field constraints (string lengths, array limits not yet enforced); no explicit payload size limit beyond Fastify defaults. Low risk for development-phase system. |
| T-03-19 | I (Information Disclosure) | GET /api/templates/:name -- internal fields | accept | WorkflowTemplate response includes eventVersion, lastPublishedVersion -- these are versioning metadata, not secrets. Harmless for the current access model. |
