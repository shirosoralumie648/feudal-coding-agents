    ```typescript
    export interface TemplateParameter {
      readonly name: string;

      readonly required: boolean;

      readonly sourceStepId: string;

      readonly id: string;
      readonly type: "intake" | "planning" | "review" | "approval" | "execution" | "verification";

      readonly dependsOn: string[];
      readonly conditions?: TemplateCondition[];
      readonly config?: Record<string, unknown>;
    }

    export type TemplateStatus = "draft" | "published";

    export interface WorkflowTemplate {
      readonly name: string;
      readonly version: string;
      readonly parameters: TemplateParameter[];
      readonly steps: TemplateStep[];

      readonly createdAt: string;
      readonly updatedAt: string;
      readonly lastPublishedVersion?: number;
      readonly eventVersion: number;
    }

    export interface TemplateInstantiation {
      readonly templateName: string;
      readonly templateVersion: string;
      readonly parameters: Record<string, string | number | boolean>;
    ```

    ```typescript
    /**
     * Recursively interpolates ${params.paramName} references in a template step's config object.
     * Per D-08: parameter interpolation using ${params.paramName} syntax.
     */
    export function interpolateParams(
      value: unknown,
      parameters: Record<string, string | number | boolean>
    ): unknown;
    ```

    Implementation rules:
    1. If `value` is a string, replace all `${params.paramName}` patterns with the corresponding value from the `parameters` record
    2. If a referenced parameter is not found in the `parameters` record, throw `new Error(\`Template parameter "\${paramName}" not provided\`)`
    3. If `value` is an array, recursively interpolate each element
    4. If `value` is a plain object, recursively interpolate each property value
    5. If `value` is a primitive (number, boolean, null, undefined), return it unchanged
    6. Replace each reference with its string representation (i.e., `String(parameters["paramName"])`)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Template JSON input (parser boundary) | External JSON template definitions cross into the type system — must be validated by Zod schemas before any engine logic consumes them |
| Parameter values (instantiation boundary) | User-provided parameter values are strings/numbers/booleans injected into step config — must be validated against template parameter definitions before interpolation |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | T (Tampering) | TemplateParameter.type field | mitigate | Zod enum validation restricts type to "string" \| "number" \| "boolean" \| "enum"; unknown types reject at parse boundary |
| T-03-02 | T (Tampering) | TemplateStep.type field | mitigate | Zod enum validation restricts to the six allowed step types; invalid step types cannot reach engine dispatch |
| T-03-03 | I (Information Disclosure) | interpolateParams | mitigate | Missing parameter reference throws explicit error instead of silently injecting "undefined" — prevents config leakage from silent failures |
| T-03-04 | E (Elevation of Privilege) | N/A (type system only, no auth) | accept | This plan defines types and schemas only; no execution or access control logic is introduced here |
</threat_model>
